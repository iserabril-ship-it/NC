// ============================================================ // CENTRO ESTANT · SGI · Backend Google Apps Script // Versión simplificada - máxima compatibilidad // ============================================================ var SHEET_NAME = "NC_Database"; var HEADERS = [ "id", "fecha", "mes", "año", "semana", "sector", "proceso", "tipo", "descripcion", "producto", "causa_raiz", "causa_cat", "impacto", "retrabajo", "parada_produccion", "recurrente", "piezas_afectadas", "accion_inmediata", "accion_correctiva", "accion_preventiva", "estado", "observaciones", "fecha_carga", "nombre_archivo" ]; // ───────────────────────────────────────────── // Obtener o crear la hoja de datos // ───────────────────────────────────────────── function getSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); var headerRange = sheet.getRange(1, 1, 1, HEADERS.length); headerRange.setBackground("#1a56db"); headerRange.setFontColor("#ffffff"); headerRange.setFontWeight("bold"); sheet.setFrozenRows(1); } return sheet; } // ───────────────────────────────────────────── // Obtener todos los registros // ───────────────────────────────────────────── function getAllRecords() { var sheet = getSheet(); var lastRow = sheet.getLastRow(); if (lastRow <= 1) { return []; } var data = sheet.getRange(2, 1, lastRow - 1, HEADERS.length).getValues(); var records = []; for (var i = 0; i < data.length; i++) { var row = data[i]; var record = {}; record._row = i + 2; for (var j = 0; j < HEADERS.length; j++) { var val = row[j]; record[HEADERS[j]] = (val !== null && val !== undefined) ? String(val) : ""; } records.push(record); } return records; } // ───────────────────────────────────────────── // Crear respuesta JSON // ───────────────────────────────────────────── function makeJSON(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ───────────────────────────────────────────── // GET - leer datos // ───────────────────────────────────────────── function doGet(e) { try { var action = "getAll"; if (e && e.parameter && e.parameter.action) { action = e.parameter.action; } if (action === "ping") { var records = getAllRecords(); return makeJSON({ status: "ok", data: { message: "Centro Estant NC Platform conectado", time: new Date().toISOString(), records: records.length } }); } if (action === "getAll") { var allRecords = getAllRecords(); var clean = []; for (var i = 0; i < allRecords.length; i++) { var r = allRecords[i]; var obj = {}; for (var j = 0; j < HEADERS.length; j++) { obj[HEADERS[j]] = r[HEADERS[j]] || ""; } clean.push(obj); } return makeJSON({ status: "ok", data: clean }); } return makeJSON({ status: "error", data: { error: "Accion no reconocida" } }); } catch (err) { return makeJSON({ status: "error", data: { error: err.toString() } }); } } // ───────────────────────────────────────────── // POST - escribir datos // ───────────────────────────────────────────── function doPost(e) { try { var body = JSON.parse(e.postData.contents); var action = body.action; if (action === "insert") { return makeJSON(insertRecord(body.data)); } if (action === "update") { return makeJSON(updateRecord(body.id, body.data)); } if (action === "delete") { return makeJSON(deleteRecord(body.id)); } if (action === "bulkInsert") { return makeJSON(bulkInsert(body.data)); } return makeJSON({ status: "error", data: { error: "Accion no reconocida" } }); } catch (err) { return makeJSON({ status: "error", data: { error: err.toString() } }); } } // ───────────────────────────────────────────── // INSERTAR un registro // ───────────────────────────────────────────── function insertRecord(data) { var sheet = getSheet(); var allRecords = getAllRecords(); // Calcular ID var maxId = 0; for (var i = 0; i < allRecords.length; i++) { var id = parseInt(allRecords[i].id) || 0; if (id > maxId) maxId = id; } data.id = maxId + 1; data.fecha_carga = new Date().toISOString(); // Calcular semana y año desde la fecha if (data.fecha) { var parts = data.fecha.split("/"); if (parts.length === 3) { try { var d = new Date(parseInt(parts[2]), parseInt(parts[1]) - 1, parseInt(parts[0])); var startYear = new Date(d.getFullYear(), 0, 1); var weekNum = Math.ceil(((d - startYear) / 86400000 + startYear.getDay() + 1) / 7); data.semana = "S" + weekNum; data.año = d.getFullYear().toString(); } catch(ex) { data.semana = ""; data.año = new Date().getFullYear().toString(); } } } // Construir fila en el orden de HEADERS var row = []; for (var i = 0; i < HEADERS.length; i++) { var val = data[HEADERS[i]]; row.push((val !== null && val !== undefined) ? val : ""); } sheet.appendRow(row); // Color por impacto var lastRow = sheet.getLastRow(); var impacto = String(data.impacto || "").toLowerCase(); var bgColor = "#ffffff"; if (impacto === "alto") bgColor = "#fff0f0"; else if (impacto === "medio") bgColor = "#fffbe6"; else if (impacto === "bajo") bgColor = "#f0faf5"; sheet.getRange(lastRow, 1, 1, HEADERS.length).setBackground(bgColor); return { status: "ok", data: { id: data.id, message: "NC registrada correctamente" } }; } // ───────────────────────────────────────────── // ACTUALIZAR un registro // ───────────────────────────────────────────── function updateRecord(id, newData) { var sheet = getSheet(); var allRecords = getAllRecords(); var record = null; for (var i = 0; i < allRecords.length; i++) { if (parseInt(allRecords[i].id) === parseInt(id)) { record = allRecords[i]; break; } } if (!record) { return { status: "error", data: { error: "Registro no encontrado" } }; } // Combinar datos existentes con nuevos var updated = {}; for (var j = 0; j < HEADERS.length; j++) { var key = HEADERS[j]; updated[key] = (newData[key] !== undefined) ? newData[key] : (record[key] || ""); } var row = []; for (var k = 0; k < HEADERS.length; k++) { row.push(updated[HEADERS[k]] || ""); } sheet.getRange(record._row, 1, 1, HEADERS.length).setValues([row]); // Recolorear var impacto = String(updated.impacto || "").toLowerCase(); var bgColor = "#ffffff"; if (impacto === "alto") bgColor = "#fff0f0"; else if (impacto === "medio") bgColor = "#fffbe6"; else if (impacto === "bajo") bgColor = "#f0faf5"; sheet.getRange(record._row, 1, 1, HEADERS.length).setBackground(bgColor); return { status: "ok", data: { message: "NC actualizada correctamente" } }; } // ───────────────────────────────────────────── // ELIMINAR un registro // ───────────────────────────────────────────── function deleteRecord(id) { var sheet = getSheet(); var allRecords = getAllRecords(); var record = null; for (var i = 0; i < allRecords.length; i++) { if (parseInt(allRecords[i].id) === parseInt(id)) { record = allRecords[i]; break; } } if (!record) { return { status: "error", data: { error: "Registro no encontrado" } }; } sheet.deleteRow(record._row); return { status: "ok", data: { message: "NC eliminada correctamente" } }; } // ───────────────────────────────────────────── // INSERTAR MÚLTIPLES registros // ───────────────────────────────────────────── function bulkInsert(records) { var count = 0; for (var i = 0; i < records.length; i++) { var result = insertRecord(records[i]); if (result.status === "ok") count++; } return { status: "ok", data: { message: count + " registros importados correctamente" } }; } // ───────────────────────────────────────────── // FUNCIÓN DE PRUEBA - podés ejecutarla manualmente // para verificar que todo funciona antes de desplegar // ───────────────────────────────────────────── function testScript() { var sheet = getSheet(); Logger.log("Hoja encontrada: " + sheet.getName()); Logger.log("Registros actuales: " + getAllRecords().length); Logger.log("Todo OK - podés desplegar como Web App"); }
